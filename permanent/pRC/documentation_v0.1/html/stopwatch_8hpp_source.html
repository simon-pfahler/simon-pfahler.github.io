23 static inline Float<64> getTimeInSeconds()
24 {
25 timespec timeSpec;
26 if(clock_gettime(CLOCK_MONOTONIC, &timeSpec))
27 {
28 Logging::error("Unable to get current time.");
29 }
30 return static_cast<double>(timeSpec.tv_nsec) / 1e9 + timeSpec.tv_sec;
31 }
