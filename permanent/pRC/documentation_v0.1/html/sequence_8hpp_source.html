692 template<class T, T N, T... Ps>
693 static inline constexpr auto integerFactorization(Constant<T, N> const,
694 Sequence<T, Ps...> const = Sequence<T>())
695 {
696 constexpr auto P = []()
697 {
698 if(N == 0)
699 {
700 return T(0);
701 }
702
703 if(N % 2 == 0)
704 {
705 return T(2);
706 }
707
708 for(T i = 3; i <= iSqrt(N); i += 2)
709 {
710 if(N % i == 0)
711 {
712 return i;
713 }
714 }
715
716 return N;
717 }();
718
719 if constexpr(P == N)
720 {
721 return Sequence<T, Ps..., N>();
722 }
723 else
724 {
725 return integerFactorization(Constant<T, N / P>(),
726 Sequence<T, Ps..., P>());
727 }
728 }
